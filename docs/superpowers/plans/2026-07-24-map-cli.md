```cpp
// Widen a visited key/value to the slot's expected integer type. Int
// literals are strictly i32 in Liva; Map<_, i64> slots would otherwise
// receive a 4-byte store into an 8-byte alloca (upper half undefined).
auto widenToSlot = [&](llvm::Value *v, llvm::Type *slotTy) -> llvm::Value * {
    if (v && slotTy && v->getType() != slotTy &&
        v->getType()->isIntegerTy() && slotTy->isIntegerTy() &&
        v->getType()->getIntegerBitWidth() < slotTy->getIntegerBitWidth())
        return builder_->CreateSExt(v, slotTy, "map.widen");
    return v;
};
```

Uygulama noktaları: Map insert (keyVal→info.keyType, valVal→info.valType), get/contains/remove (keyVal→info.keyType), Set insert/contains/remove (val→info.elemType). Store'dan HEMEN önce genişlet.

```liva
// cli::cli — Command-line argument parsing
//
//   var p = ArgParser.new("myapp", "Does things")
//   p.addFlag("verbose", "v", "Verbose output")
//   p.addOption("out", "o", "out.txt", "Output file")
//   p.addPositional("input", "Input file")
//   let r = p.parse(os.getArgs())
//   if !r.ok { println(r.error); println(p.usage()); return }

pub struct ParseResult {
    pub var ok: bool
    pub var error: string
    pub var helpRequested: bool
    var setFlags: [string]
    var optNames: [string]
    var optValues: [string]
    var posValues: [string]
}

impl ParseResult {
    pub func getFlag(ref self, long: string) -> bool {
        for f in self.setFlags {
            if f == long { return true }
        }
        return false
    }
    pub func getOption(ref self, long: string) -> string {
        var i = 0
        for n in self.optNames {
            if n == long { return self.optValues[i] }
            i = i + 1
        }
        return ""
    }
    pub func positionalCount(ref self) -> i64 {
        return self.posValues.length
    }
    pub func getPositional(ref self, i: i64) -> string? {
        if i < 0 { return nil }
        if i >= self.posValues.length { return nil }
        return self.posValues[i]
    }
}

pub struct ArgParser {
    var name: string
    var description: string
    var flagLongs: [string]
    var flagShorts: [string]
    var flagHelps: [string]
    var optLongs: [string]
    var optShorts: [string]
    var optDefaults: [string]
    var optHelps: [string]
    var posNames: [string]
    var posHelps: [string]
}

impl ArgParser {
    pub func new(name: string, description: string) -> ArgParser {
        return ArgParser {
            name: name, description: description,
            flagLongs: [], flagShorts: [], flagHelps: [],
            optLongs: [], optShorts: [], optDefaults: [], optHelps: [],
            posNames: [], posHelps: []
        }
    }
    pub func addFlag(ref mut self, long: string, short: string, help: string) { ... }
    pub func addOption(ref mut self, long: string, short: string, defaultVal: string, help: string) { ... }
    pub func addPositional(ref mut self, name: string, help: string) { ... }
    pub func parse(ref self, args: [string]) -> ParseResult { ... }
}
```

parse algoritması (indeksli while döngüsü — lookahead için for-in yerine `var i = 0; while i < args.length`):
1. `arg == "--"` → kalanların hepsi posValues'a, bitir.
2. `arg == "--help" || arg == "-h"` → helpRequested=true, devam.
3. `strStartsWith(arg, "--")`: `strContains(arg, "=")` ise `strSplit(arg, "=")` ile böl (ad = ilk parça `--` soyulmuş: `name[2..len]` slicing veya substring), değeri kaydet; değilse ad çöz → flag ise setFlags'e; option ise sonraki arg'ı değer al (`i+1 >= args.length` veya sonraki `strStartsWith(next, "-")` → hata "missing value for --X"); ikisi de değilse hata `"unknown option: --X"`.
4. `strStartsWith(arg, "-")` ve `arg != "-"`: kısa ad tek karakter varsay (`arg[1..arg.length]`), flagShorts/optShorts'ta ara; eşleşme yoksa hata `"unknown option: -X"`.
5. Aksi halde posValues'a ekle.
Bitişte: helpRequested değilse `posValues.length < posNames.length` → hata `"missing required positional: <ad>"`. Option tekrarında optNames'te varsa değeri GÜNCELLE (son kazanır — optValues[i] üzerine yaz; dizi eleman ataması `self.optValues[i] = v`). Hata durumunda `ok=false, error=mesaj` ve o ana kadarki sonuçla dön (ilk hatada durur).
Hata mesajları testlerdeki literallere birebir: `"unknown option: --unknown"` biçimi.
